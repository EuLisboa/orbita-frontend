Outras funções do app.js
// Copiar texto Orbita.copiar('https://orbita.app') // Formatar número: 3420 → "3.420" Orbita.format.numero(3420) // Formatar tempo relativo Orbita.format.tempo(new Date()) // "agora mesmo" // Validar e-mail Orbita.validar.email('teste@gmail.com') // true // Força da senha Orbita.validar.senha('Senha@123') // → { forca: 4, nivel: 'forte' }
Como criar uma nova página
<!-- 1. No <head>, importe as fontes e o CSS --> <link href="https://fonts.googleapis.com/css2?family=Syne:wght@400;700;800&family=DM+Sans:wght@300;400;500&display=swap" rel="stylesheet"/> <link rel="stylesheet" href="styles.css"/> <!-- 2. Seu CSS específico vem depois --> <style> /* só o que não existe no styles.css */ </style> <!-- 3. Antes do </body>, importe o JS --> <script src="app.js"></script> <!-- 4. Seu JS específico vem depois --> <script> // use Orbita.toast(), Orbita.modal, etc. </script>
